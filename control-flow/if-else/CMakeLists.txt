cmake_minimum_required(VERSION 3.14)

if(FALSE)
	message("if false block executed")
elseif(TRUE)
	message("elseif true block executed")
else()
	message("else block executed")
endif()

set(FOO OFF)
set(BAR FOO)

if(DEFINED FOO)
	message("FOO: if block executed")
	set(BIZ FOO)
else()
	message("FOO: else block executed")
endif()

if(${BAR})
	message("BAR: if block executed")
else()
	message("BAR: else block executed")
endif()

if(DEFINED BAZ)
	message("BAZ: if block executed")
else()
	message("BAZ: else block executed")
endif()

if(DEFINED BIZ)
message("BIZ: if block executed")
else()
	message("BIZ: else block executed")
endif()

if(EXISTS CMakeLists.txt)
	message("CMakeLists.txt exists")
else()
	message("CMakeLists.txt not found")
endif()
